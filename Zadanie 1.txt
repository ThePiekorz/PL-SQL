CREATE SEQUENCE OSOBY_SEQ
 	START WITH 1
 	INCREMENT BY 1
	NOCACHE
	NOCYCLE;

CREATE OR REPLACE TRIGGER OSOBY_TRG
	BEFORE INSERT ON OSOBY
FOR EACH ROW
DECLARE
 	y NUMBER;
BEGIN
 	IF :NEW.ID_OSOBY IS NULL OR :NEW.ID_OSOBY = 0 THEN
		SELECT OSOBY_SEQ.NEXTVAL INTO y FROM DUAL;
		:NEW.ID_OSOBY := y;
 	END IF;
END;
/
